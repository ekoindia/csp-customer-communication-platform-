# ── LITE install profile — for the real 4 GB deploy PC (Dell Inspiron 3268) ──
#
# This machine (4 GB RAM, i3-7100, no GPU) runs in Tesseract-only OCR mode
# (core/hardware.py auto-selects it below OCR_RAM_THRESHOLD_GB), so PyTorch +
# docTR are NEVER loaded there. Installing them would waste ~2 GB of the ~52 GB
# free disk and add a heavy, slow install for zero benefit. So this profile
# deliberately OMITS `python-doctr` and does NOT install torch/torchvision.
#
#   pip install -r requirements-lite.txt
#
# The scanned-document path still works (grid detection via OpenCV + Tesseract
# word reading). CSV/Excel/typed-PDF need no OCR at all. If this box is ever
# upgraded to >= 6 GB RAM and you want docTR accuracy, install the full
# requirements.txt instead (adds python-doctr + CPU torch) — no code change.
#
# Tesseract-OCR itself is a separate Windows install (core/ocr.py looks for it
# at C:\Program Files\Tesseract-OCR\tesseract.exe).

flask==3.0.3
pydantic==2.7.4
openpyxl==3.1.2
pdfplumber==0.11.1
pillow==10.3.0
pytesseract==0.3.10
requests==2.32.3
# (pytest is NOT a runtime dependency — it lives in requirements-dev.txt so the
#  slow 4 GB deploy box doesn't download/install it. Developers: pip install -r
#  requirements-dev.txt.)
# Loads local secrets/config from a .env file (never committed — see
# .env.example) instead of hardcoding them in config.py.
python-dotenv==1.0.1
# Field-level encryption at rest for customer PII (core/crypto.py) — required
# on every profile, lite included, since PII protection can't depend on OCR mode.
cryptography==43.0.3
# Scanned-document OCR support (grid detection + page rendering), no deep learning:
pypdfium2==4.30.0
numpy==2.1.3

psutil==5.9.8
# Runtime for the custom on-device digit OCR model (core/ocr_onnx.py). CPU-only,
# ~15-50 MB, numpy-2 compatible — NOT PyTorch. Lets the trained crnn.onnx read
# account/mobile digits accurately on the 4 GB CSP box. Falls back to Tesseract
# if absent, so the app still runs without it.
onnxruntime==1.19.2
